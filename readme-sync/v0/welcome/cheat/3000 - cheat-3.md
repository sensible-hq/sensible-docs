Checkboxes and regions
===

Solution: Mix it up
---

After you've completed all the Mix it up steps in the tutorial, the final config is:

```json
{
  "fields": [
    {
      "id": "preference_tues_fruit",
      "anchor": {
        "match": [
          {
            "type": "includes",
            "text": "tuesday"
          },
          {
            "type": "includes",
            "text": "fruit"
          }
        ]
      },
      "method": {
        "id": "checkbox",
        "position": "left"
      }
    },
    {
      "id": "preference_wed_fruit",
      "anchor": {
        "match": [
          {
            "type": "includes",
            "text": "wednesday"
          },
          {
            "type": "includes",
            "text": "fruit"
          }
        ]
      },
      "method": {
        "id": "checkbox",
        "position": "left"
      }
    },
    {
      "id": "tuesday_all_options",
      "anchor": "tuesday",
      "method": {
        "id": "region",
        "start": "below",
        "width": 6.8,
        "height": 0.5,
        "offsetX": -0.5,
        "offsetY": 0
      }
    },
    {
      "id": "disclaimer_paragraph_and_more",
      "anchor": {
        "match": {
          "type": "startsWith",
          "text": "please note"
        }
      },
      "method": {
        "id": "documentRange",
        "includeAnchor": true,
        "stop": {
          "type": "startsWith",
          "text": "mix it up"
        }
      }
    }
  ]
}
```
